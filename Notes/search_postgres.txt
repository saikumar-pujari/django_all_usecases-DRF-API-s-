from django.shortcuts import render
from django.contrib.postgres.search import SearchQuery, SearchRank
from .models import Blog

def search_blog(request):

    search_text = request.GET.get("q", "").strip()

    blogs = Blog.objects.none()

    if search_text:

        query = SearchQuery(search_text)

        blogs = (
            Blog.objects
            .annotate(
                rank=SearchRank("search_vector", query)
                //(only if weights are specified)   rank=SearchRank("search_vector", query,weights=[0.1, 0.2, 0.4, 1.0])
                //postgres assigns weights in this order: [D, C, B, A] only!!
            )
            .filter(
                search_vector=query,
                rank__gte=0.1
            )
            .order_by("-rank")
        )

    return render(
        request,
        "search.html",
        {
            "blogs": blogs,
            "search_text": search_text
        }
    )

from django.contrib.postgres.search import (
    SearchQuery,
    SearchRank,
    TrigramSimilarity,
)
query = SearchQuery(search_text)
blogs = (
    Blog.objects
    .annotate(
        rank=SearchRank("search_vector", query),
        similarity=TrigramSimilarity("title", search_text),
    )
    .filter(
        Q(search_vector=query) |
        Q(similarity__gt=0.3)
    )
    .order_by("-rank", "-similarity")
)
